## Implementation Notes

### Data Structure for Database

```typescript
interface Brand {
  id: string
  name: string
  origin: string
  website: string
  categories: string[]
  positioning: 'economy' | 'mid-range' | 'premium' | 'luxury'
  alumimundoStatus?: string
}

interface ProductCategory {
  id: string
  brandId: string
  name: string
  nameEs: string
  subcategories?: string[]
}

interface Product {
  id: string
  sku: string
  brandId: string
  categoryId: string
  name: string
  nameEs?: string
  description: string
  descriptionEs?: string
  specifications: {
    material?: string
    dimensions?: string
    weight?: string
    warranty?: string
    certifications?: string[]
    finishes?: string[]
    colors?: string[]
  }
  pricing: {
    currency: 'USD' | 'CRC'
    price?: number
    priceRange?: { min: number; max: number }
  }
  images: {
    primary: string
    gallery?: string[]
  }
  availability: {
    inStock: boolean
    leadTime?: string
  }
  tags: string[]
  catalogUrl?: string
}
```
